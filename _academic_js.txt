
function toggleCustomFormat(){document.getElementById('customFormatBox').style.display=document.querySelector('input[name="acFormat"]:checked').value==='Personalizado'?'block':'none';}
function parseAuthors(input){if(!input||!input.trim())return[];return input.split(/\s*;\s*/).map(function(a){var p=a.split(/\s*,\s*/);return{surname:(p[0]||'').trim(),given:(p[1]||'').trim(),initials:(p[1]||'').trim().split(/\s+/).map(function(w){return w.charAt(0).toUpperCase()+'.';}).join('')};}).filter(function(a){return a.surname;});}
function formatNames(authors,fmt){if(!authors.length)return'[Autor]';if(fmt==='MLA'){var f=authors[0],r=f.surname+', '+f.given;if(authors.length>1)r+=authors.length===2?', y '+authors[1].given:' et al.';return r;}if(fmt==='APA'||fmt==='Harvard')return authors.map(function(a,i){var s=a.surname+', '+a.initials;if(authors.length>1&&i===authors.length-1)s=' & '+s;else if(i>0)s=', '+s;return s;}).join('');if(fmt==='Vancouver')return authors.map(function(a,i){var s=a.surname+' '+a.initials.replace(/\./g,'');if(i<authors.length-1)s+=',';return s;}).join(' ');if(fmt==='Turabian'||fmt==='Chicago')return authors.map(function(a,i){var s=a.surname+', '+a.given;if(authors.length>1&&i===authors.length-1)s=', y '+s;else if(i>0&&i<authors.length-1)s+=',';return s;}).join('');if(fmt==='IEEE')return authors.map(function(a,i){var s=a.initials+' '+a.surname;if(authors.length>1&&i===authors.length-1)s=', y '+s;else if(i>0)s+=',';return s;}).join('');return authors.map(function(a){return a.surname+', '+a.initials;}).join('; ');}

'+title+'. <em>'+journal+'</em>'+(volume?', '+volume:'')+(issue?'('+issue+')':'')+(pages?', '+pages:'')+'.'+(doi?' https://doi.org/'+doi:'');}else if(fmt==='MLA'){cite=mlaNames+'. "'+title+'." <em>'+journal+'</em>, vol. '+(volume||'X')+', no. '+(issue||'X')+', '+year+(pages?', pp. '+pages:'')+'.'+(doi?' doi:'+doi:'')+'.';}else if(fmt==='Chicago'){cite=turabianNames+'. '+year+'. "'+title+'." <em>'+journal+'</em> '+(volume||'')+(issue?' ('+issue+'):':': ')+(pages||'')+'.'+(doi?' https://doi.org/'+doi:'');}else if(fmt==='Harvard'){cite=harvardNames+' ('+year+') \''+title+'\', <em>'+journal+'</em>, '+volume+'('+issue+'), pp. '+pages+'. doi:'+doi+'.';}else if(fmt==='Vancouver'){cite=vancouverNames+'. '+title+'. '+journal+'. '+year+';'+(volume||'')+(issue?'('+issue+'):':':')+(pages||'')+'.'+(doi?' doi:'+doi:'');}else if(fmt==='Turabian'){cite=turabianNames+'. "'+title+'." <em>'+journal+'</em> '+volume+', no. '+issue+' ('+year+'): '+pages+'.'+(doi?' doi:'+doi:'');}else if(fmt==='IEEE'){cite=ieeeNames+', "'+title+'," <em>'+journal+'</em>, vol. '+volume+', no. '+issue+(pages?', p. '+pages:'')+(year?', '+year:'')+(month?', '+month:'')+(day?' '+day:'')+'. [Abstract].'+(doi?' Available: https://doi.org/'+doi:'')+' [Accessed '+nowAccess+'].';}else if(fmt==='Personalizado'){var tpl=document.getElementById('acCustomTpl').value.trim();if(!tpl)tpl='{APA_NAMES} ({YEAR}). {TITLE}. {JOURNAL}, {VOLUME}({ISSUE}), {PAGES}.';cite=tpl.replace(/{APA_NAMES}/g,apaNames).replace(/{MLA_NAMES}/g,mlaNames).replace(/{TURABIAN_NAMES}/g,turabianNames).replace(/{HARVARD_NAMES}/g,harvardNames).replace(/{VANCOUVER_NAMES}/g,vancouverNames).replace(/{IEEE_NAMES}/g,ieeeNames).replace(/{TITLE}/g,title).replace(/{JOURNAL}/g,journal).replace(/{YEAR}/g,year).replace(/{MONTH}/g,month).replace(/{DAY}/g,day).replace(/{VOLUME}/g,volume).replace(/{ISSUE}/g,issue).replace(/{PAGES}/g,pages).replace(/{DOI}/g,doi).replace(/{NOW}/g,nowAccess);}document.getElementById('acOutput').innerHTML=cite;}
